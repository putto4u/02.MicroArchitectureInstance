# 리눅스 시스템 명령어 모음

>데몬  
sudo systemctl start nginx  
sudo systemctl stop nginx  
sudo systemctl restart nginx  
sudo systemctl reload nginx  


sudo ss -tunlp    # 포트를 점유하고 있는 프로세스 ID(PID)와 프로그램 명 확인
또는  sudo netstat -plntu  # 또는 netstat 사용 (net-tools 패키지 필요)


ps -ef --forest    # 실행 중인 모든 프로세스를 트리 형태로 확인
ps -ef | grep ssh    # 특정 데몬이 실행 중인지 검색 (예: ssh)

---


sudo lsof -i :80    # 특정 포트를 사용 중인 데몬 확인 (예: 80번 포트)
